# react-xterm-prototype

## Install

```bash
npm install --save react-xterm-prototype
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-xterm-prototype'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
